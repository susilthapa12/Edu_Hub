Merging & Joining DataFrames

Pandas can merge and join multiple datasets efficiently using merge and join.

import pandas as pd

df1 = pd.DataFrame({'ID':[1,2,3], 'Name':['Alice','Bob','Charlie']})
df2 = pd.DataFrame({'ID':[2,3,4], 'Salary':[50000,60000,70000]})

merged_inner = pd.merge(df1, df2, on='ID', how='inner')
merged_left = pd.merge(df1, df2, on='ID', how='left')

print(merged_inner)
print(merged_left)

Concatenating DataFrames

df3 = pd.DataFrame({'ID':[5,6], 'Name':['David','Eva']})
concatenated = pd.concat([df1, df3], ignore_index=True)
print(concatenated)

Pivot Tables

data = {'Department':['HR','IT','IT','HR'],
        'Employee':['Alice','Bob','Charlie','David'],
        'Salary':[50000,60000,70000,55000]}

df = pd.DataFrame(data)
pivot = df.pivot_table(index='Department', values='Salary', aggfunc='mean')
print(pivot)

Filtering & Querying

high_salary = df.query('Salary > 55000')
print(high_salary)

it_employees = df[(df['Department'] == 'IT') & (df['Salary'] > 60000)]
print(it_employees)

Advanced Visualization

import matplotlib.pyplot as plt
import seaborn as sns

df.groupby('Department')['Salary'].mean().plot(kind='bar', color='skyblue')
plt.title('Average Salary by Department')
plt.ylabel('Salary')
plt.show()

sns.countplot(x='Department', data=df)
plt.show()
