MySQL Indexes

Indexes improve database performance by speeding up searches on a table.

-- Create an index on LastName column
CREATE INDEX idx_lastname ON Students(LastName);

-- Create a unique index on Email column
CREATE UNIQUE INDEX idx_email ON Students(Email);

MySQL Views

Views are virtual tables based on a SELECT query. They simplify complex queries.

-- Create a view of students in 10th grade
CREATE VIEW TenthGradeStudents AS
SELECT FirstName, LastName, Grade
FROM Students
WHERE Grade = '10th';

-- Query the view
SELECT * FROM TenthGradeStudents;

Stored Procedures

Stored procedures are reusable SQL code that can be executed repeatedly.

-- Create a stored procedure to get students by grade
DELIMITER //
CREATE PROCEDURE GetStudentsByGrade(IN gradeName VARCHAR(10))
BEGIN
    SELECT * FROM Students WHERE Grade = gradeName;
END //
DELIMITER ;

-- Call the procedure
CALL GetStudentsByGrade('10th');
