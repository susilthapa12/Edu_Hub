Django Models

Models define the structure of your database using Python classes.

Model Example

from django.db import models

class Book(models.Model):
    title = models.CharField(max_length=200)
    author = models.CharField(max_length=100)
    published_date = models.DateField()
    price = models.DecimalField(max_digits=6, decimal_places=2)

    def __str__(self):
        return self.title

Django ORM Queries

Django ORM allows interaction with the database using Python.

Basic Queries

# Get all books
books = Book.objects.all()

# Filter by author
books_by_author = Book.objects.filter(author='John Doe')

# Get single book
book = Book.objects.get(id=1)

Create, Update, Delete

# Create a new book
new_book = Book.objects.create(
    title='Python 101',
    author='Alice',
    published_date='2025-01-01',
    price=29.99
)

# Update a book
book_to_update = Book.objects.get(id=1)
book_to_update.price = 24.99
book_to_update.save()

# Delete a book
book_to_delete = Book.objects.get(id=2)
book_to_delete.delete()

Advanced Django Features

Relationships

class Author(models.Model):
    name = models.CharField(max_length=100)

class Book(models.Model):
    title = models.CharField(max_length=200)
    author = models.ForeignKey(Author, on_delete=models.CASCADE)

Model Meta Options

class Book(models.Model):
    title = models.CharField(max_length=200)

    class Meta:
        ordering = ['title']
        verbose_name = 'Book'
        verbose_name_plural = 'Books'

Custom Model Methods

class Book(models.Model):
    title = models.CharField(max_length=200)

    def is_long_title(self):
        return len(self.title) > 50
